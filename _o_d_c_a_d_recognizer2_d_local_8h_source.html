62  /** \brief Simple RANSAC based 3D object recognizer.
63  *
64  * A recognizer which uses local features like SIFT/SURF to perform object recognition.
65  * Given a 'trained model' trained by ODCADRecogTrainerSnapshotBased or trained externally (manually augmenting features in 3D cad models), this class performs a complete detection
66  * in an image. It first extracts 2D features from the scene, matches them with all the feature augmented models (the trained data) and in the end solves PnP under RANSAC.
67  *
68  * \author Kripasindhu Sarkar
69  *
70  */

194  ODCADRecognizer2DLocal(string const &trained_data_location_ = 0) : ODImageLocalMatchingDetector(
195  trained_data_location_)
196  {
197  metainfo_ = true;
198 
199  camera_intrinsic_file = "Data/out_camera_data_lion_old.yml"; // mesh
200 
201  red = cv::Scalar(0, 0, 255);
202  green = cv::Scalar(0, 255, 0);
203  blue = cv::Scalar(255, 0, 0);
204  yellow = cv::Scalar(0, 255, 255);
205 
206 
207  numKeyPoints = 2000; // number of detected keypoints
208  ratioTest = 0.70f; // ratio test
209  fast_match = true; // fastRobustMatch() or robustMatch()
210  use_gpu = false;
211  use_gpu_match = false;
212 
213  iterationsCount = 500; // number of Ransac iterations.
214  reprojectionError = 2.0; // maximum allowed distance to consider it an inlier.
215  confidence = 0.95; // ransac successful confidence.
216 
217  minInliers = 30; // Kalman threshold updating
218 
219  pnpMethod = cv::SOLVEPNP_EPNP;
220  f_type_default = "SIFT";
221  featureDetector = boost::make_shared<ODFeatureDetector2D>(f_type_default, use_gpu);
222  }
