100  {
101  ODDetections3D *detections = new ODDetections3D;
102 
103 
104  typename pcl::PointCloud<PointT>::Ptr frame;
105  float Z_DIST_ = 1.25f;
106  float text_scale = 0.015f;
107 
108  frame = scene->getPointCloud();
109  pcl::PointCloud<pcl::PointXYZ>::Ptr xyz_points(new pcl::PointCloud<pcl::PointXYZ>);
110  pcl::copyPointCloud(*frame, *xyz_points);
111 
112 
113  //Step 1 -> Segment
114  pcl::apps::DominantPlaneSegmentation<pcl::PointXYZ> dps;
115  dps.setInputCloud(xyz_points);
116  dps.setMaxZBounds(Z_DIST_);
117  dps.setObjectMinHeight(0.005);
118  dps.setMinClusterSize(1000);
119  dps.setWSize(9);
120  dps.setDistanceBetweenClusters(0.1f);
121 
122  std::vector<pcl::PointCloud<pcl::PointXYZ>::Ptr> clusters;
123  std::vector<pcl::PointIndices> indices;
124  dps.setDownsamplingSize(0.02f);
125  dps.compute_fast(clusters);
126  dps.getIndicesClusters(indices);
127  Eigen::Vector4f table_plane_;
128  Eigen::Vector3f normal_plane_ = Eigen::Vector3f(table_plane_[0], table_plane_[1], table_plane_[2]);
129  dps.getTableCoefficients(table_plane_);
130 
131 
132  float dist_ = 0.03f;
133 
134  for(size_t i = 0; i < clusters.size(); i++)
135  {
136 
137  global_->setInputCloud(xyz_points);
138  global_->setIndices(indices[i].indices);
139  global_->classify();
140 
141  std::vector<std::string> categories;
142  std::vector<float> conf;
143  global_->getCategory(categories);
144  global_->getConfidence(conf);
145  //detection done!
146 
147  std::string category = categories[0];
148  Eigen::Vector4d centroid;
149  pcl::compute3DCentroid(*xyz_points, indices[i].indices, centroid);
150  //position at 3D identified!
151 
152  //now fill up the detection:
153  ODDetection3D *detection = new ODDetection3D;
154  detection->setType(ODDetection::OD_DETECTION_CLASS);
155  detection->setId(categories[0]);
156  detection->setLocation(centroid);
157  detection->setMetainfoCluster(clusters[i]);
158  detections->push_back(detection);
159  }
