43  /** \brief The main detector class; all special Detectors derives from this.
44  *
45  * Provides interface for two important function detect() and detectOmni(). detectOmni() performs a detection/recognition on the entire scene (unsegmented and unprocessed)
46  * and provides information about the detection as well as its exact location. detect() takes an 'object candidate' or a segmented/processed scene as an input and identifies if the entire scene is a detection.
47  *
48  * \author Kripasindhu Sarkar
49  *
50  */
51  class ODDetector: public ODDetectorCommon
52  {
53  public:
54 
55  ODDetector(std::string const &training_data_location_) : ODDetectorCommon(training_data_location_)
56  { }
57 
58  virtual ODDetections* detect(ODScene *scene){}
59  virtual ODDetections* detectOmni(ODScene *scene){}
60 
61  bool metainfo_;
62 
63  };

65 /** \brief The detector of 2D scene.
66  *
67  * This class takes a 2D scene (ODSceneImage) as input and performs detection on them. All the 2D detectors should derive from this class and implement the detect and detectOmni functions.
68  *
69  * \author Kripasindhu Sarkar
70  *
71  */
72  class ODDetector2D: public ODDetector
73  {
74  public:
75  ODDetector2D(std::string const &trained_data_location_) : ODDetector(trained_data_location_)
76  { }
77 
78  ODDetections* detect(ODScene *scene)
79  {
80  return detect(dynamic_cast<ODSceneImage *>(scene));
81  }
82 
83  /** \brief Function for performing detection on a segmented scene.
84  * The purpose of this function is to perform detection on a segmented scene or an 'object candidate'. i.e. the entire scene is considered as an 'object' or an detection. It is possible for a scene to trigger multiple detections.
85  * \param[in] scene An instance of 2D scene
86  * \return [out] detections A number of detections as an ODDetections instance.
87  */
88  virtual ODDetections* detect(ODSceneImage *scene) = 0;
89 
90  /** \brief Function for performing detection on an entire scene.
91  * The purpose of this function is to detect an object in an entire scene. Thus, other than the type of detection we also have information about the location of the detection w.r.t. the scene.
92  * \param[in] scene An instance of 2D scene
93  * \return [out] detections A number of detections as an ODDetections2D instance containing information about the detection and its 2D location.
94  */
95  virtual ODDetections2D* detectOmni(ODSceneImage *scene) = 0;
96  };
